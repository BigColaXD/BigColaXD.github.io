# ElGamal

使用多个不同的 QNR 作为m，或者对一个 QNR 做变换生成更多QNR。

若a ∈ QR, b ∈ QNR⇒a × b ∈ QNR

所以我们可以取：m\_i = (i^2) \* base\_qnr % p，其中base\_qnr是一个固定的QNR

因为i^2 ∈ QR，乘上 QNR 还是 QNR ⇒ 所有m\_i ∈ QNR

这样就能生成无限多个不同的QNR明文

```python
from pwn import *

p = 218065457789864386530000483395774968164217154628843306669369488732001946982207

def is_qr(x, p):
    return pow(x, (p-1)//2, p) == 1

def find_qnr(p):
    """找最小的 QNR"""
    a = 2
    while True:
        if not is_qr(a, p):
            return a
        a += 1

# ========== 开始连接 ==========
context.log_level = 'info'

io = remote("111.74.9.131", 18063)

# 接收 pub 行
line = io.recvline().strip().decode()
print(f"[+] Raw: {line}")

# 解析 pub
try:
    if '=' in line:
        pub = int(line.split('=', 1)[1].strip())
    else:
        pub = int(line.strip())
except Exception as e:
    print(f"[-] Parse pub failed: {e}")
    io.close()
    exit(1)

success(f"✅ Pub = {pub}")

# 找一个基础 QNR
base_qnr = find_qnr(p)
success(f"✅ Base QNR = {base_qnr}")

correct_count = 0

try:
    for i in range(1, 1338):  # 最多 1337 轮
        info(f"🔁 Round {i}")

        # 构造新的 QNR: m = (i^2) * base_qnr mod p
        m = (pow(i, 2, p) * base_qnr) % p
        debug(f"Generated m = {m} (QR={is_qr(m, p)})")

        # 发送 m
        try:
            io.sendlineafter(b'm>', str(m).encode(), timeout=2)
        except:
            io.sendline(str(m).encode())

        # 接收密文
        try:
            cipher_line = io.recvline(timeout=2).decode().strip()
            debug(f"Cipher raw: {cipher_line}")

            if not cipher_line.startswith('(') or ')' not in cipher_line:
                warning("Invalid format")
                break

            inner = cipher_line.strip('() ')
            c1_str, c2_str = [x.strip() for x in inner.split(',', 1)]
            c1 = int(c1_str)
            c2 = int(c2_str)

        except Exception as e:
            warning(f"Parse ciphertext error: {e}")
            break

        # 判断是否为真实密文（c1 ∈ QR, c2 ∈ QNR）
        c1_is_qr = is_qr(c1, p)
        c2_is_qnr = not is_qr(c2, p)

        if c1_is_qr and c2_is_qnr:
            guess = 1
        else:
            guess = 0

        # 回答 coin
        try:
            io.sendlineafter(b'coin>', str(guess).encode(), timeout=2)
        except:
            io.sendline(str(guess).encode())

        # 检查反馈
        try:
            response = io.recvline(timeout=2).decode().strip()
            debug(f"Response: {response}")
            if "Correct" in response:
                correct_count += 1
                success(f"✅ Correct! Total: {correct_count}")
            else:
                info("❌ Wrong")
        except:
            warning("Timeout reading response")
            break

    # 尝试读取 flag
    try:
        flag_data = io.recvall(timeout=3).decode().strip()
        if flag_data:
            success(f"🚩 FLAG FOUND:\n{flag_data}")
    except:
        pass

except Exception as e:
    warning(f"Error: {e}")
finally:
    io.close()
```
