# 命令

```powershell
#REG ADD KeyName(这里使用 FullKey 全路径用于区分) /v xxx(所选项之下要添加的值名称) /t REG_SZ(值的类型) /d xxx(值内容) /f

# HKEY_LOCAL_MACHINE下
reg add HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows\CurrentVersion\Run /v Reg1 /t REG_SZ /d "C:\Users\Administrator\Desktop\regAutoRun2.exe" /f
# HKEY_CURRENT_USER下
reg add HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Run /v Reg2 /t REG_SZ /d "C:\Users\Administrator\Desktop\regAutoRun1.exe" /f

# 查询
reg query HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows\CurrentVersion\Run /f Reg1
# 删除
reg delete HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows\CurrentVersion\Run /v Reg1 /f
reg delete HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Run /v Reg2 /f
```
